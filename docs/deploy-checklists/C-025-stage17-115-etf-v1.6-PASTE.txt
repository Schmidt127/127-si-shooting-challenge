function getFieldSafe(table, fieldName) {
  if (!table || !fieldName) return null;
  const cacheKey = `${table.name}:${fieldName}`;
  if (fieldCache.has(cacheKey)) return fieldCache.get(cacheKey);
  try {
    const field = table.getField(fieldName);
    fieldCache.set(cacheKey, field);
    return field;
  } catch {
    fieldCache.set(cacheKey, null);
    return null;
  }
}

function fieldExists(table, fieldName) {
  return !!getFieldSafe(table, fieldName);
}

function requireField(table, fieldName) {
  if (!fieldExists(table, fieldName)) {
    throw new Error(`Missing required field on ${table.name}: ${fieldName}`);
  }
}

function isWritableField(table, fieldName) {
  const field = getFieldSafe(table, fieldName);
  if (!field) return false;
  if (field.isComputed === true) return false;
  const nonWritableTypes = new Set([
    "formula",
    "rollup",
    "count",
    "lookup",
    "multipleLookupValues",
    "createdTime",
    "lastModifiedTime",
    "createdBy",
    "lastModifiedBy",
    "autoNumber",
    "button",
    "aiText",
    "externalSyncSource",
  ]);
  return !nonWritableTypes.has(field.type);
}

function addIfWritable(table, payload, fieldName, value) {
  if (!fieldName || !fieldExists(table, fieldName)) return;
  if (!isWritableField(table, fieldName)) {
    console.log(`Skipped non-writable field: ${table.name} -> ${fieldName}`);
    return;
  }
  if (value === undefined) return;
  payload[fieldName] = value;
}

function buildSafeUpdatePayload(table, fields) {
  const safeFields = {};
  for (const [fieldName, value] of Object.entries(fields || {})) {
    addIfWritable(table, safeFields, fieldName, value);
  }
  return safeFields;
}

function getText(record, fieldName) {
  if (!fieldName) return "";
  return String(record.getCellValueAsString(fieldName) || "").trim();
}

function getNumber(record, fieldName) {
  if (!fieldName) return 0;
  const value = record.getCellValue(fieldName);
  if (typeof value === "number") return value;
  const parsed = Number(value);
  return Number.isFinite(parsed) ? parsed : 0;
}

function getBooleanish(record, fieldName, fallback = false) {
  if (!fieldName) return fallback;
  const value = record.getCellValue(fieldName);
  if (value === true) return true;
  if (value === false) return false;
  if (value === 1) return true;
  if (value === 0) return false;
  const text = String(value ?? "").trim().toLowerCase();
  return ["1", "true", "yes", "checked"].includes(text);
}

function getLinkedIds(record, fieldName) {
  if (!fieldName) return [];
  const value = record.getCellValue(fieldName);
  if (!Array.isArray(value)) return [];
  return value.map((item) => item.id).filter(Boolean);
}

function getDateValue(record, fieldName) {
  if (!fieldName) return null;
  const value = record.getCellValue(fieldName);
  if (!value) return null;
  if (value instanceof Date && !isNaN(value)) return value;
  if (typeof value === "string") {
    const parsed = new Date(value);
    return isNaN(parsed) ? null : parsed;
  }
  return null;
}

function getAttachments(record, fieldName) {
  if (!fieldName || !record) return [];
  try {
    const value = record.getCellValue(fieldName);
    return Array.isArray(value) ? value : [];
  } catch {
    return [];
  }
}

function attachmentFilename(file) {
  return String(file?.filename || file?.name || "").trim();
}

function listAttachmentFilenames(files) {
  return (files || []).map(attachmentFilename).filter(Boolean);
}

function getTestingScenarioVideoAttachmentFieldNames(table) {
  return CONFIG.testingScenarioVideoAttachmentFields.filter((fieldName) =>
    fieldExists(table, fieldName)
  );
}

function getVideoAttachmentsFromScenario(record, table) {
  const candidateFields = getTestingScenarioVideoAttachmentFieldNames(table);
  let best = { files: [], sourceField: "", candidateFields };

  for (const fieldName of candidateFields) {
    const files = getAttachments(record, fieldName);
    if (files.length > best.files.length) {
      best = { files, sourceField: fieldName, candidateFields };
    }
  }

  return best;
}

function buildAttachmentWritePayload(files) {
  return (files || [])
    .filter((file) => file && file.url)
    .map((file) => ({
      url: file.url,
      filename: file.filename || "uploaded_file",
    }));
}

function singleSelectValue(table, fieldName, choiceName) {
  const field = getFieldSafe(table, fieldName);
  if (!field) return undefined;
  if (field.type === "singleSelect") {
    const choices = field.options?.choices || [];
    if (!choices.some((choice) => choice.name === choiceName)) {
      throw new Error(
        `Missing single-select option "${choiceName}" on ${table.name} -> ${fieldName}.`
      );
    }
    return { name: choiceName };
  }
  return choiceName;
}

async function updateTestingScenario(testingScenariosTable, scenarioId, fields) {
  const safeFields = buildSafeUpdatePayload(testingScenariosTable, fields);
  if (Object.keys(safeFields).length > 0) {
    await testingScenariosTable.updateRecordAsync(scenarioId, safeFields);
  }
}

async function finishScenarioRun({
  testingScenariosTable,
  scenarioId,
  scenarioType,
  debugStep,
  actionOut,
  statusOut,
  errorOut,
  createdSubmissionId,
  createdRecordSummary,
  runMetadata,
  clearRunTest,
}) {
  const updateFields = {
    [CONFIG.testingScenarioFields.lastRunAt]: new Date(),
    ...runMetadata,
  };
  if (clearRunTest) {
    updateFields[CONFIG.testingScenarioFields.runTest] = false;
  }
  await updateTestingScenario(testingScenariosTable, scenarioId, updateFields);

  setOutputSafe("statusOut", statusOut);
  setOutputSafe("actionOut", actionOut);
  setOutputSafe("errorOut", errorOut || "");
  setOutputSafe("debugStep", debugStep);
  setOutputSafe("testingScenarioIdOut", scenarioId);
  setOutputSafe("createdSubmissionIdOut", createdSubmissionId || "");
  setOutputSafe("scenarioTypeOut", scenarioType || "");
  setOutputSafe("createdRecordSummaryOut", createdRecordSummary || "");
}

async function blockScenarioRun({
  testingScenariosTable,
  scenarioId,
  scenarioType,
  debugStep,
  actionOut,
  errorOut,
  actualResult,
  passFailNotes,
  statusOut = CONFIG.statuses.skipped,
  lastRunStatus = CONFIG.lastRunStatuses.blocked,
}) {
  await finishScenarioRun({
    testingScenariosTable,
    scenarioId,
    scenarioType,
    debugStep,
    actionOut,
    statusOut,
    errorOut,
    createdSubmissionId: "",
    createdRecordSummary: "",
    clearRunTest: true,
    runMetadata: {
      [CONFIG.testingScenarioFields.lastRunStatus]: singleSelectValue(
        testingScenariosTable,
        CONFIG.testingScenarioFields.lastRunStatus,
        lastRunStatus
      ),
      [CONFIG.testingScenarioFields.actualResult]: actualResult,
      [CONFIG.testingScenarioFields.passFailNotes]: passFailNotes || errorOut,
    },
  });
}

async function runDailySubmissionBranch({
  testingScenariosTable,
  submissionsTable,
  scenarioRecord,
  recordId,
  enrollmentId,
  athleteId,
  activityDate,
  isDryRun,
  debugStep,
}) {
  validateDailySubmissionSchema({
    testingScenarios: testingScenariosTable,
    submissions: submissionsTable,
  });

  const shotTotal = getNumber(scenarioRecord, CONFIG.testingScenarioFields.shotTotal);
  if (!shotTotal || shotTotal <= 0) {
    await blockScenarioRun({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioType: CONFIG.scenarioTypes.dailySubmission,
      debugStep,
      actionOut: CONFIG.actions.skippedMissingInput,
      errorOut: "Shot Total must be a positive number on Testing Scenarios.",
      actualResult:
        "Missing or invalid Shot Total. Add Testing Scenarios.Shot Total on DEV.",
    });
    return;
  }

  if (isDryRun) {
    const preview = buildDailyDryRunPreview({
      scenarioRecord,
      enrollmentId,
      athleteId,
      activityDate,
      shotTotal,
    });

    await finishScenarioRun({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioType: CONFIG.scenarioTypes.dailySubmission,
      debugStep: "5 - Daily dry run preview",
      actionOut: CONFIG.actions.dryRun,
      statusOut: CONFIG.statuses.success,
      errorOut: "",
      createdSubmissionId: "",
      createdRecordSummary: `Dry run — Daily Submission; Shot Total ${shotTotal}.`,
      clearRunTest: true,
      runMetadata: {
        [CONFIG.testingScenarioFields.lastRunStatus]: singleSelectValue(
          testingScenariosTable,
          CONFIG.testingScenarioFields.lastRunStatus,

/* =========================================================
   SECTION 5B — C-025 STAGE 17 DOWNSTREAM (057 + 042) — v1.6
   Query-budgeted: exact-record reads only, capped polls (5×),
   resume-safe, never writes Attendees. Daily/HW/Video unchanged.
   042 trigger: view `042` re-entry via Level Recalc Needed?
   (checked→unchecked→checked, or unchecked→checked).
========================================================= */

const C025_MAX_QUERY_BUDGET = 22;

function delayMs(ms) {
  // Airtable automation scripts do not support real async timers (see 070b v4.4).
  // Busy-wait is intentional so capped polls actually space out before re-select.
  const end = Date.now() + Math.max(0, Number(ms) || 0);
  while (Date.now() < end) {
    /* busy-wait */
  }
}

function createC025QueryBudget(max) {
  const log = [];
  let count = 0;
  return {
    max,
    get count() {
      return count;
    },
    get log() {
      return log.slice();
    },
    consume(purpose) {
      if (count + 1 > max) {
        const err = new Error(
          `C025 query budget exhausted (${count}/${max}) before: ${purpose}`
        );
        err.code = "C025_QUERY_BUDGET";
        throw err;
      }
      count += 1;
      log.push({ n: count, purpose });
      console.log(JSON.stringify({ c025Query: count, max, purpose }));
      return count;
    },
  };
}

function isC025Stage17DownstreamScenario(scenarioType, testIntakeName, scenarioRequirements) {
  if (
    scenarioType !== CONFIG.scenarioTypes.other &&
    scenarioType !== CONFIG.scenarioTypes.perfectWeek
  ) {
    return false;
  }
  const blob = `${testIntakeName || ""}\n${scenarioRequirements || ""}`;
  return blob.includes(CONFIG.c025Stage17Downstream.scenarioKey);
}

function parseC025FixtureOverrides(scenarioRequirementsText) {
  const text = String(scenarioRequirementsText || "").trim();
  if (!text) return {};
  try {
    const parsed = JSON.parse(text);
    if (!parsed || typeof parsed !== "object") return {};
    return {
      enrollmentId: parsed.enrollmentId || parsed.enrollment || undefined,
      weekId: parsed.weekId || parsed.week || undefined,
      zoomAttendanceId: parsed.zoomAttendanceId || parsed.zaId || undefined,
      zoomMeetingId: parsed.zoomMeetingId || parsed.meetingId || undefined,
      wasId: parsed.wasId || parsed.was || undefined,
      resetFixtures: parsed.resetFixtures === true,
    };
  } catch {
    return {};
  }
}

function resolveC025Fixtures(scenarioRequirementsText) {
  return {
    enrollmentId: CONFIG.c025Stage17Downstream.enrollmentId,
    weekId: CONFIG.c025Stage17Downstream.weekId,
    zoomAttendanceId: CONFIG.c025Stage17Downstream.zoomAttendanceId,
    zoomMeetingId: CONFIG.c025Stage17Downstream.zoomMeetingId,
    wasId: CONFIG.c025Stage17Downstream.wasId,
    resetFixtures: false,
    ...parseC025FixtureOverrides(scenarioRequirementsText),
  };
}

function refuseAttendeesWrite(fields) {
  if (!fields || typeof fields !== "object") return;
  if (Object.prototype.hasOwnProperty.call(fields, CONFIG.zoomMeetingFields.attendees)) {
    throw new Error("Refuse Attendees write — C025 must never modify Zoom Meetings.Attendees");
  }
}

async function updateRecordNoAttendees(table, recordId, fields) {
  refuseAttendeesWrite(fields);
  const safe = buildSafeUpdatePayload(table, fields);
  if (Object.keys(safe).length > 0) {
    await table.updateRecordAsync(recordId, safe);
  }
}

async function c025SelectRecord(budget, table, recordId, fields, purpose) {
  budget.consume(purpose);
  return table.selectRecordAsync(recordId, { fields });
}

async function runC025Stage17DownstreamBranch({
  testingScenariosTable,
  enrollmentsTable,
  recordId,
  enrollmentId,
  isDryRun,
  debugStep,
  scenarioRequirements,
}) {
  const scenarioTypeLabel = CONFIG.c025Stage17Downstream.scenarioKey;
  const fixtures = resolveC025Fixtures(scenarioRequirements);
  const budget = createC025QueryBudget(
    CONFIG.c025Stage17Downstream.maxQueryBudget || C025_MAX_QUERY_BUDGET
  );
  const pollAttempts = CONFIG.c025Stage17Downstream.pollAttempts || 5;
  const pollDelayMs = CONFIG.c025Stage17Downstream.pollDelayMs || 2500;

  let currentStep = "C025-0-start";
  let attempts057 = 0;
  let attempts042 = 0;
  let resumedFrom = "fresh";

  const finishAwaiting = async (phase, notes, extra) => {
    await finishC025Run({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioTypeLabel,
      debugStep: currentStep,
      actionOut:
        phase === "057" ? CONFIG.actions.c025Awaiting057 : CONFIG.actions.c025Awaiting042,
      statusOut: CONFIG.statuses.skipped,
      errorOut: notes,
      lastRunStatus: CONFIG.lastRunStatuses.blocked,
      actualPayload: {
        scenario: scenarioTypeLabel,
        status: phase === "057" ? "Timed Out Waiting for 057" : "Timed Out Waiting for 042",
        queryCount: budget.count,
        queryBudget: budget.max,
        queryLog: budget.log,
        currentStep,
        attempts057,
        attempts042,
        fixtures,
        resumedFrom,
        ...extra,
      },
      passFailNotes: `${notes} Re-check Run Test? later to resume. Turn 057/042 OFF when idle.`,
    });
  };

  try {
    if (enrollmentId !== fixtures.enrollmentId) {
      await blockScenarioRun({
        testingScenariosTable,
        scenarioId: recordId,
        scenarioType: scenarioTypeLabel,
        debugStep,
        actionOut: CONFIG.actions.skippedInvalidEnrollment,
        errorOut: `C025 requires Related Enrollment ${fixtures.enrollmentId}. Found ${enrollmentId}.`,
        actualResult: "Enrollment mismatch for C025_STAGE17_DOWNSTREAM.",
      });
      return;
    }

    const wasTable = base.getTable(CONFIG.tables.weeklyAthleteSummary);
    const zaTable = base.getTable(CONFIG.tables.zoomAttendance);
    const zmTable = base.getTable(CONFIG.tables.zoomMeetings);

    const zaFields = [
      CONFIG.zoomAttendanceFields.pwApplied,
      CONFIG.zoomAttendanceFields.gateApplied,
      CONFIG.zoomAttendanceFields.approved,
      CONFIG.zoomAttendanceFields.conflict,
      CONFIG.zoomAttendanceFields.pwFlag,
      CONFIG.zoomAttendanceFields.gateEarned,
      CONFIG.zoomAttendanceFields.reviewStatus,
      CONFIG.zoomAttendanceFields.enrollment,
      CONFIG.zoomAttendanceFields.zoomMeeting,
      CONFIG.zoomAttendanceFields.attendanceMethod,
    ].filter((n) => fieldExists(zaTable, n));

    const wasFields = [
      CONFIG.wasFields.automationStatus,
      CONFIG.wasFields.automationError,
      CONFIG.wasFields.dailyMet,
      CONFIG.wasFields.videoCount,
      CONFIG.wasFields.zoomMeetingCount,
      CONFIG.wasFields.zoomAttendanceCount,
      CONFIG.wasFields.zoomRequirementMet,
      CONFIG.wasFields.zoomRequirementStatus,
      CONFIG.wasFields.goalRecord,
    ].filter((n) => fieldExists(wasTable, n));

    const enrFields = [
      CONFIG.enrollmentFields.levelRecalcNeeded,
      CONFIG.enrollmentFields.currentLevel,
      CONFIG.enrollmentFields.nextLevel,
      CONFIG.enrollmentFields.levelStatus,
      CONFIG.enrollmentFields.levelGateRule,
      CONFIG.enrollmentFields.totalZoomAttendances,
    ].filter((n) => fieldExists(enrollmentsTable, n));

    const zmFields = [CONFIG.zoomMeetingFields.attendees, CONFIG.zoomMeetingFields.week].filter(
      (n) => fieldExists(zmTable, n)
    );

    currentStep = "C025-1-initial-snapshot";
    setOutputSafe("debugStep", currentStep);

    let zaRec = await c025SelectRecord(
      budget,
      zaTable,
      fixtures.zoomAttendanceId,
      zaFields,
      "initial ZA"
    );
    let wasRec = await c025SelectRecord(budget, wasTable, fixtures.wasId, wasFields, "initial WAS");
    let zmRec = await c025SelectRecord(
      budget,
      zmTable,
      fixtures.zoomMeetingId,
      zmFields,
      "initial ZM"
    );
    let enrRec = await c025SelectRecord(
      budget,
      enrollmentsTable,
      enrollmentId,
      enrFields,
      "initial Enrollment"
    );

    if (!zaRec || !wasRec || !zmRec || !enrRec) {
      await blockScenarioRun({
        testingScenariosTable,
        scenarioId: recordId,
        scenarioType: scenarioTypeLabel,
        debugStep: currentStep,
        actionOut: CONFIG.actions.skippedMissingInput,
        errorOut: "One or more C025 fixture records were not found.",
        actualResult: JSON.stringify({ fixtures, queryCount: budget.count }, null, 2),
      });
      return;
    }

    const attendeesBefore = getLinkedIds(zmRec, CONFIG.zoomMeetingFields.attendees);
    let pwApplied = getBooleanish(zaRec, CONFIG.zoomAttendanceFields.pwApplied, false);
    let gateApplied = getBooleanish(zaRec, CONFIG.zoomAttendanceFields.gateApplied, false);
    const pwAppliedBefore = pwApplied;
    const gateAppliedBefore = gateApplied;
    const currentLevelBefore = getLinkedIds(enrRec, CONFIG.enrollmentFields.currentLevel)[0] || "";
    const nextLevelBefore = getLinkedIds(enrRec, CONFIG.enrollmentFields.nextLevel)[0] || "";
    const levelStatusBefore = getText(enrRec, CONFIG.enrollmentFields.levelStatus);
    const dailyMetBefore = getBooleanish(wasRec, CONFIG.wasFields.dailyMet, false);
    const videoCountBefore = getNumber(wasRec, CONFIG.wasFields.videoCount);

    const qualifying =
      getText(zaRec, CONFIG.zoomAttendanceFields.attendanceMethod) === "Recording Quiz" &&
      getBooleanish(zaRec, CONFIG.zoomAttendanceFields.approved, false) &&
      !getBooleanish(zaRec, CONFIG.zoomAttendanceFields.conflict, false) &&
      getBooleanish(zaRec, CONFIG.zoomAttendanceFields.pwFlag, false) &&
      getBooleanish(zaRec, CONFIG.zoomAttendanceFields.gateEarned, false) &&
      getText(zaRec, CONFIG.zoomAttendanceFields.reviewStatus) !== "Needs Correction";

    if (!qualifying) {
      await blockScenarioRun({
        testingScenariosTable,
        scenarioId: recordId,
        scenarioType: scenarioTypeLabel,
        debugStep: currentStep,
        actionOut: CONFIG.actions.skippedMissingInput,
        errorOut: "Zoom Attendance fixture is not qualifying for recording PW + gate credit.",
        actualResult: JSON.stringify({ fixtures, queryCount: budget.count }, null, 2),
        lastRunStatus: CONFIG.lastRunStatuses.fail,
        statusOut: CONFIG.statuses.error,
      });
      return;
    }

    if (fixtures.resetFixtures) {
      resumedFrom = "explicit_reset";
      await updateRecordNoAttendees(zaTable, fixtures.zoomAttendanceId, {
        [CONFIG.zoomAttendanceFields.pwApplied]: false,
        [CONFIG.zoomAttendanceFields.gateApplied]: false,
      });
      pwApplied = false;
      gateApplied = false;
    } else if (pwApplied && gateApplied) {
      resumedFrom = "resume_both_done";
    } else if (pwApplied) {
      resumedFrom = "resume_after_057";
    } else {
      resumedFrom = "fresh";
    }

    if (isDryRun) {
      await finishC025Run({
        testingScenariosTable,
        scenarioId: recordId,
        scenarioTypeLabel,
        debugStep: "C025-dry-run",
        actionOut: CONFIG.actions.dryRun,
        statusOut: CONFIG.statuses.success,
        errorOut: "",
        lastRunStatus: CONFIG.lastRunStatuses.pass,
        actualPayload: {
          scenario: scenarioTypeLabel,
          dryRun: true,
          fixtures,
          queryCount: budget.count,
          queryBudget: budget.max,
          resumedFrom,
          attendeesBefore,
          trigger057: "WAS status Ready→Pending (skipped if PW Applied already)",
          trigger042:
            "View 042 re-entry: Level Recalc Needed? checked→unchecked→checked (or unchecked→checked); skipped if Gate Applied already",
          neverWrite: ["Attendees"],
          pollAttempts,
        },
        passFailNotes:
          "Dry run only — no triggers. Uncheck Dry Run? for live run with 057/042 ON.",
      });
      return;
    }

    // ---------- Phase A: 057 ----------
    let phaseA = {
      skipped: false,
      waitOk: false,
      attempts: 0,
      timedOut: false,
    };

    if (pwApplied) {
      currentStep = "C025-2-skip-057-already-applied";
      setOutputSafe("debugStep", currentStep);
      phaseA.skipped = true;
      phaseA.waitOk = true;
      phaseA.pwAppliedBefore = pwAppliedBefore;
      phaseA.pwAppliedAfter = true;
    } else {
      currentStep = "C025-2-trigger-057";
      setOutputSafe("debugStep", currentStep);
      await updateRecordNoAttendees(wasTable, fixtures.wasId, {
        [CONFIG.wasFields.automationStatus]: singleSelectValue(
          wasTable,
          CONFIG.wasFields.automationStatus,
          "Ready"
        ),
      });
      await delayMs(400);
      await updateRecordNoAttendees(wasTable, fixtures.wasId, {
        [CONFIG.wasFields.automationStatus]: singleSelectValue(
          wasTable,
          CONFIG.wasFields.automationStatus,
          "Pending"
        ),
        [CONFIG.wasFields.automationError]: "",
      });

      currentStep = "C025-2-poll-057";
      setOutputSafe("debugStep", currentStep);
      const poll057Fields = [
        CONFIG.zoomAttendanceFields.pwApplied,
        CONFIG.zoomAttendanceFields.gateApplied,
      ].filter((n) => fieldExists(zaTable, n));
      for (let i = 0; i < pollAttempts; i += 1) {
        attempts057 += 1;
        await delayMs(pollDelayMs);
        zaRec = await c025SelectRecord(
          budget,
          zaTable,
          fixtures.zoomAttendanceId,
          poll057Fields,
          `poll057 ZA #${attempts057}`
        );
        if (getBooleanish(zaRec, CONFIG.zoomAttendanceFields.pwApplied, false)) {
          phaseA.waitOk = true;
          pwApplied = true;
          gateApplied = getBooleanish(zaRec, CONFIG.zoomAttendanceFields.gateApplied, false);
          break;
        }
      }
      phaseA.attempts = attempts057;
      phaseA.pwAppliedBefore = pwAppliedBefore;
      phaseA.pwAppliedAfter = pwApplied;

      if (!phaseA.waitOk) {
        phaseA.timedOut = true;
        await finishAwaiting("057", "Timed Out Waiting for 057", {
          phaseA_057: phaseA,
          attendeesBefore,
        });
        return;
      }
    }

    // ---------- Phase B: 042 (view re-entry) ----------
    let phaseB = {
      skipped: false,
      waitOk: false,
      attempts: 0,
      timedOut: false,
      viewReentry042: null,
    };

    if (gateApplied) {
      currentStep = "C025-3-skip-042-already-applied";
      setOutputSafe("debugStep", currentStep);
      phaseB.skipped = true;
      phaseB.waitOk = true;
      phaseB.gateAppliedBefore = gateAppliedBefore;
      phaseB.gateAppliedAfter = true;
      phaseB.viewReentry042 = {
        skip: true,
        transition: "skip_already_applied",
        reason: "Gate Credit Applied? already true — skip 042 retrigger",
        triggerOnce: false,
      };
    } else {
      currentStep = "C025-3-trigger-042-view-reentry";
      setOutputSafe("debugStep", currentStep);

      const recalcReadFields = [CONFIG.enrollmentFields.levelRecalcNeeded].filter((n) =>
        fieldExists(enrollmentsTable, n)
      );
      enrRec = await c025SelectRecord(
        budget,
        enrollmentsTable,
        enrollmentId,
        recalcReadFields,
        "pre042 Level Recalc Needed?"
      );
      const levelRecalcPrevious = getBooleanish(
        enrRec,
        CONFIG.enrollmentFields.levelRecalcNeeded,
        false
      );

      // Force leave/re-enter view `042` (When record enters view; condition: Level Recalc Needed? checked).
      // Writing checked→checked alone does NOT re-enter the view.
      const viewReentry = {
        view: "042",
        viewCondition: "Level Recalc Needed? is checked",
        enrollmentId,
        previousLevelRecalcNeeded: levelRecalcPrevious,
        resetValue: null,
        armedValue: null,
        viewEntryTriggeredAt: null,
        transition: null,
        triggerOnce: true,
        confirmedUnchecked: null,
      };

      if (levelRecalcPrevious) {
        viewReentry.transition = "checked → unchecked → checked";
        viewReentry.resetValue = false;
        await updateRecordNoAttendees(enrollmentsTable, enrollmentId, {
          [CONFIG.enrollmentFields.levelRecalcNeeded]: false,
        });
        delayMs(400);
        enrRec = await c025SelectRecord(
          budget,
          enrollmentsTable,
          enrollmentId,
          recalcReadFields,
          "confirm042 Level Recalc unchecked"
        );
        const confirmedUnchecked = !getBooleanish(
          enrRec,
          CONFIG.enrollmentFields.levelRecalcNeeded,
          false
        );
        viewReentry.confirmedUnchecked = confirmedUnchecked;
        if (!confirmedUnchecked) {
          throw new Error(
            "042 view re-entry failed: Level Recalc Needed? did not clear after uncheck write"
          );
        }
        delayMs(400);
        await updateRecordNoAttendees(enrollmentsTable, enrollmentId, {
          [CONFIG.enrollmentFields.levelRecalcNeeded]: true,
        });
        viewReentry.armedValue = true;
      } else {
        viewReentry.transition = "unchecked → checked";
        viewReentry.resetValue = null;
        await updateRecordNoAttendees(enrollmentsTable, enrollmentId, {
          [CONFIG.enrollmentFields.levelRecalcNeeded]: true,
        });
        viewReentry.armedValue = true;
      }
      viewReentry.viewEntryTriggeredAt = new Date().toISOString();
      phaseB.viewReentry042 = viewReentry;
      console.log(
        JSON.stringify({
          c025PhaseB042: "view_reentry_armed",
          ...viewReentry,
          queryCount: budget.count,
        })
      );

      currentStep = "C025-3-poll-042";
      setOutputSafe("debugStep", currentStep);
      for (let i = 0; i < pollAttempts; i += 1) {
        attempts042 += 1;
        delayMs(pollDelayMs);
        zaRec = await c025SelectRecord(
          budget,
          zaTable,
          fixtures.zoomAttendanceId,
          [CONFIG.zoomAttendanceFields.gateApplied].filter((n) => fieldExists(zaTable, n)),
          `poll042 ZA #${attempts042}`
        );
        if (getBooleanish(zaRec, CONFIG.zoomAttendanceFields.gateApplied, false)) {
          phaseB.waitOk = true;
          gateApplied = true;
          break;
        }
      }
      phaseB.attempts = attempts042;
      phaseB.gateAppliedBefore = gateAppliedBefore;
      phaseB.gateAppliedAfter = gateApplied;

      if (!phaseB.waitOk) {
        phaseB.timedOut = true;
        await finishAwaiting("042", "Timed Out Waiting for 042", {
          phaseA_057: phaseA,
          phaseB_042: phaseB,
          attendeesBefore,
        });
        return;
      }
    }

    // ---------- Final verify + dedupe (no re-trigger; targeted reads only) ----------
    currentStep = "C025-4-final-verify";
    setOutputSafe("debugStep", currentStep);

    wasRec = await c025SelectRecord(budget, wasTable, fixtures.wasId, wasFields, "final WAS");
    zaRec = await c025SelectRecord(budget, zaTable, fixtures.zoomAttendanceId, zaFields, "final ZA");
    zmRec = await c025SelectRecord(budget, zmTable, fixtures.zoomMeetingId, zmFields, "final ZM");
    enrRec = await c025SelectRecord(
      budget,
      enrollmentsTable,
      enrollmentId,
      enrFields,
      "final Enrollment"
    );

    const zoomAttendanceCount = getNumber(wasRec, CONFIG.wasFields.zoomAttendanceCount);
    const zoomMeetingCount = getNumber(wasRec, CONFIG.wasFields.zoomMeetingCount);
    const attendeesAfter = getLinkedIds(zmRec, CONFIG.zoomMeetingFields.attendees);
    const pwFinal = getBooleanish(zaRec, CONFIG.zoomAttendanceFields.pwApplied, false);
    const gateFinal = getBooleanish(zaRec, CONFIG.zoomAttendanceFields.gateApplied, false);
    const recalcFinal = getBooleanish(enrRec, CONFIG.enrollmentFields.levelRecalcNeeded, false);
    const currentLevelAfter = getLinkedIds(enrRec, CONFIG.enrollmentFields.currentLevel)[0] || "";
    const nextLevelAfter = getLinkedIds(enrRec, CONFIG.enrollmentFields.nextLevel)[0] || "";
    const levelStatusAfter = getText(enrRec, CONFIG.enrollmentFields.levelStatus);

    const attendeesUnchanged =
      JSON.stringify(attendeesBefore) === JSON.stringify(attendeesAfter) &&
      attendeesAfter.length === 0;

    const passA =
      Number(zoomAttendanceCount) === 1 &&
      Number(zoomMeetingCount) >= 1 &&
      pwFinal === true &&
      attendeesUnchanged;
    // Prefer Gate Applied; when we triggered 042, also expect Level Recalc cleared by workflow.
    // Resume-skip (already Applied) does not require recalc cleared.
    const passB =
      gateFinal === true &&
      attendeesUnchanged &&
      (phaseB.skipped === true || recalcFinal === false);
    const passDedupe =
      Number(zoomAttendanceCount) === 1 && pwFinal && gateFinal && attendeesAfter.length === 0;
    const overallPass = passA && passB && passDedupe && attendeesUnchanged;

    if (recalcFinal) {
      await updateRecordNoAttendees(enrollmentsTable, enrollmentId, {
        [CONFIG.enrollmentFields.levelRecalcNeeded]: false,
      });
    }

    const resultPayload = {
      scenario: scenarioTypeLabel,
      queryCount: budget.count,
      queryBudget: budget.max,
      queryLog: budget.log,
      currentStep,
      attempts057,
      attempts042,
      resumedFrom,
      fixtures,
      attendeesBefore,
      attendeesAfter,
      attendeesUnchanged,
      phaseA_057: {
        ...phaseA,
        dailyMetBefore,
        videoCountBefore,
        zoomMeetingCount,
        zoomAttendanceCount,
        pwAppliedFinal: pwFinal,
      },
      phaseB_042: {
        ...phaseB,
        currentLevelBefore,
        nextLevelBefore,
        levelStatusBefore,
        currentLevelAfter,
        nextLevelAfter,
        levelStatusAfter,
        levelRecalcNeededAfter: recalcFinal,
        gateAppliedFinal: gateFinal,
        diagnostics: {
          previousLevelRecalcNeeded: phaseB.viewReentry042
            ? phaseB.viewReentry042.previousLevelRecalcNeeded
            : null,
          resetValue: phaseB.viewReentry042 ? phaseB.viewReentry042.resetValue : null,
          armedValue: phaseB.viewReentry042 ? phaseB.viewReentry042.armedValue : null,
          viewEntryTriggeredAt: phaseB.viewReentry042
            ? phaseB.viewReentry042.viewEntryTriggeredAt
            : null,
          attempts042,
          queryCount: budget.count,
          gateAppliedFinal: gateFinal,
          currentLevelBefore,
          currentLevelAfter,
          nextLevelBefore,
          nextLevelAfter,
        },
      },
      dedupe: {
        mode: "verify_without_retrigger",
        zoomAttendanceCount,
        pwApplied: pwFinal,
        gateApplied: gateFinal,
        attendees: attendeesAfter,
        pass: passDedupe,
      },
      passA,
      passB,
      passDedupe,
      overallPass,
      safety: {
        neverWroteAttendees: true,
        noFullTableSelects: true,
        automation117NotUsed: true,
        automation042ViewReentry: true,
      },
    };

    currentStep = "C025-5-write-results";
    setOutputSafe("debugStep", currentStep);

    await finishC025Run({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioTypeLabel,
      debugStep: currentStep,
      actionOut: overallPass ? CONFIG.actions.c025Pass : CONFIG.actions.c025Fail,
      statusOut: overallPass ? CONFIG.statuses.success : CONFIG.statuses.error,
      errorOut: overallPass
        ? ""
        : [
            !passA ? "Phase A (057) verify failed" : null,
            !passB ? "Phase B (042) verify failed" : null,
            !passDedupe ? "Dedupe verify failed" : null,
            !attendeesUnchanged ? "Attendees changed" : null,
          ]
            .filter(Boolean)
            .join("; "),
      lastRunStatus: overallPass ? CONFIG.lastRunStatuses.pass : CONFIG.lastRunStatuses.fail,
      actualPayload: resultPayload,
      passFailNotes: overallPass
        ? `C025_STAGE17_DOWNSTREAM PASS (queries ${budget.count}/${budget.max}). Turn 057/042 OFF.`
        : `C025_STAGE17_DOWNSTREAM FAIL (queries ${budget.count}/${budget.max}). See Actual Result.`,
    });

    console.log(
      JSON.stringify({
        automation: SCRIPT.scriptName,
        version: SCRIPT.version,
        scenario: scenarioTypeLabel,
        overallPass,
        queryCount: budget.count,
        testingScenarioId: recordId,
      })
    );
  } catch (error) {
    const message = error instanceof Error ? error.message : String(error);
    const isBudget = error && error.code === "C025_QUERY_BUDGET";
    await finishC025Run({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioTypeLabel,
      debugStep: currentStep,
      actionOut: isBudget ? CONFIG.actions.c025QueryBudget : CONFIG.actions.error,
      statusOut: CONFIG.statuses.error,
      errorOut: message,
      lastRunStatus: CONFIG.lastRunStatuses.error,
      actualPayload: {
        scenario: scenarioTypeLabel,
        status: isBudget ? "Query Budget Exhausted" : "Error",
        queryCount: budget ? budget.count : null,
        queryBudget: budget ? budget.max : C025_MAX_QUERY_BUDGET,
        queryLog: budget ? budget.log : [],
        currentStep,
        attempts057,
        attempts042,
        resumedFrom,
        fixtures,
        error: message,
      },
      passFailNotes: `C025 error — Run Test? cleared. ${message}`,
    });
  }
}


/* =========================================================
   SECTION 6 — MAIN
========================================================= */

async function main() {
  let debugStep = "0 - Start";
  setOutputSafe("debugStep", debugStep);

  const inputConfig = input.config();
  const recordId = String(inputConfig.recordId || "").trim();

  debugStep = "1 - Validate recordId";
  setOutputSafe("debugStep", debugStep);

  if (!recordId) {
    throw new Error("Missing input variable: recordId");
  }
  if (!recordId.startsWith("rec")) {
    throw new Error(`Invalid Testing Scenarios recordId input: ${recordId}`);
  }

  const testingScenariosTable = base.getTable(CONFIG.tables.testingScenarios);
  const enrollmentsTable = base.getTable(CONFIG.tables.enrollments);
  const submissionsTable = base.getTable(CONFIG.tables.submissions);

  debugStep = "2 - Validate common schema";
  setOutputSafe("debugStep", debugStep);
  validateCommonSchema({
    testingScenarios: testingScenariosTable,
    enrollments: enrollmentsTable,
    submissions: submissionsTable,
  });

  debugStep = "3 - Load Testing Scenarios row";
  setOutputSafe("debugStep", debugStep);

  const scenarioFieldsToLoad = [
    CONFIG.testingScenarioFields.testIntakeName,
    CONFIG.testingScenarioFields.scenarioType,
    CONFIG.testingScenarioFields.relatedEnrollment,
    CONFIG.testingScenarioFields.submissionDate,
    CONFIG.testingScenarioFields.shotTotal,
    CONFIG.testingScenarioFields.homeworkAssignment,
    CONFIG.testingScenarioFields.intakeAttachments,
    CONFIG.testingScenarioFields.videoFeedbackFocus,
    CONFIG.testingScenarioFields.videoFeedbackQuestion,
    CONFIG.testingScenarioFields.scenarioRequirements,
    CONFIG.testingScenarioFields.runTest,
    CONFIG.testingScenarioFields.dryRun,
    CONFIG.testingScenarioFields.linkedSubmission,
    CONFIG.testingScenarioFields.actualResult,
    CONFIG.testingScenarioFields.passFailNotes,
  ]
    .concat(getTestingScenarioVideoAttachmentFieldNames(testingScenariosTable))
    .filter((fieldName, index, all) => fieldExists(testingScenariosTable, fieldName) && all.indexOf(fieldName) === index);

  const scenarioRecord = await testingScenariosTable.selectRecordAsync(recordId, {
    fields: scenarioFieldsToLoad,
  });

  if (!scenarioRecord) {
    throw new Error(`Testing Scenarios record not found: ${recordId}`);
  }

  const scenarioType = getText(scenarioRecord, CONFIG.testingScenarioFields.scenarioType);
  const testIntakeName = fieldExists(testingScenariosTable, CONFIG.testingScenarioFields.testIntakeName)
    ? getText(scenarioRecord, CONFIG.testingScenarioFields.testIntakeName)
    : "";
  const scenarioRequirements = getText(
    scenarioRecord,
    CONFIG.testingScenarioFields.scenarioRequirements
  );
  setOutputSafe("scenarioTypeOut", normalizeScenarioTypeForOutput(scenarioType));

  // C-025 Stage 17 downstream — routed before intake Scenario Type allowlist.
  if (isC025Stage17DownstreamScenario(scenarioType, testIntakeName, scenarioRequirements)) {
    const enrollmentIdsC025 = getLinkedIds(
      scenarioRecord,
      CONFIG.testingScenarioFields.relatedEnrollment
    );
    const enrollmentIdC025 = enrollmentIdsC025[0] || "";
    if (!enrollmentIdC025) {
      await blockScenarioRun({
        testingScenariosTable,
        scenarioId: recordId,
        scenarioType: CONFIG.c025Stage17Downstream.scenarioKey,
        debugStep,
        actionOut: CONFIG.actions.skippedMissingInput,
        errorOut: "Related Enrollment is required.",
        actualResult: "Missing Related Enrollment.",
      });
      return;
    }
    if (!isAllowedEnrollmentId(enrollmentIdC025)) {
      await blockScenarioRun({
        testingScenariosTable,
        scenarioId: recordId,
        scenarioType: CONFIG.c025Stage17Downstream.scenarioKey,
        debugStep,
        actionOut: CONFIG.actions.skippedInvalidEnrollment,
        errorOut: `Enrollment ${enrollmentIdC025} is not in MVP allowlist.`,
        actualResult: "Enrollment not allowed for MVP.",
      });
      return;
    }
    const isDryRunC025 = getBooleanish(scenarioRecord, CONFIG.testingScenarioFields.dryRun, false);
    await runC025Stage17DownstreamBranch({
      testingScenariosTable,
      enrollmentsTable,
      recordId,
      enrollmentId: enrollmentIdC025,
      isDryRun: isDryRunC025,
      debugStep,
      scenarioRequirements,
    });
    return;
  }

  if (!SUPPORTED_SCENARIO_TYPES.has(scenarioType)) {
    await finishScenarioRun({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioType,
      debugStep,
      actionOut: CONFIG.actions.skippedWrongScenario,
      statusOut: CONFIG.statuses.skipped,
      errorOut: `v1.4 supports Daily Submission, Homework, Video/Three Video Upload, and C025_STAGE17_DOWNSTREAM (Scenario Type Other or Perfect Week with that name/requirements marker). Found: "${scenarioType || "blank"}".`,
      createdSubmissionId: "",
      createdRecordSummary: "",
      clearRunTest: true,
      runMetadata: {
        [CONFIG.testingScenarioFields.lastRunStatus]: singleSelectValue(
          testingScenariosTable,
          CONFIG.testingScenarioFields.lastRunStatus,
          CONFIG.lastRunStatuses.blocked
        ),
        [CONFIG.testingScenarioFields.actualResult]: "Skipped — scenario type not supported in v1.4.",
        [CONFIG.testingScenarioFields.passFailNotes]:
          `Use Daily Submission, Homework, Video, or Other/Perfect Week named ${CONFIG.c025Stage17Downstream.scenarioKey}.`,
      },
    });
    return;
  }

  const enrollmentIds = getLinkedIds(
    scenarioRecord,
    CONFIG.testingScenarioFields.relatedEnrollment
  );
  const enrollmentId = enrollmentIds[0] || "";

  if (!enrollmentId) {
    await blockScenarioRun({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioType,
      debugStep,
      actionOut: CONFIG.actions.skippedMissingInput,
      errorOut: "Related Enrollment is required.",
      actualResult: "Missing Related Enrollment.",
    });
    return;
  }

  if (!isAllowedEnrollmentId(enrollmentId)) {
    await blockScenarioRun({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioType,
      debugStep,
      actionOut: CONFIG.actions.skippedInvalidEnrollment,
      errorOut: `Enrollment ${enrollmentId} is not in MVP allowlist. Schmidt only: ${CONFIG.allowedEnrollmentIds.join(", ")}.`,
      actualResult: "Enrollment not allowed for MVP.",
      passFailNotes:
        "Use Schmidt test enrollment recgP9qZYjAhE7NXm. Expanded allowlist deferred.",
    });
    return;
  }

  const activityDate = getDateValue(scenarioRecord, CONFIG.testingScenarioFields.submissionDate);
  if (!activityDate) {
    await blockScenarioRun({
      testingScenariosTable,
      scenarioId: recordId,
      scenarioType,
      debugStep,
      actionOut: CONFIG.actions.skippedMissingInput,
      errorOut: "Submission Date is required.",
      actualResult: "Missing Submission Date.",
    });
    return;
  }

  debugStep = "4 - Load Enrollment and resolve Athlete";
  setOutputSafe("debugStep", debugStep);

  const enrollmentRecord = await enrollmentsTable.selectRecordAsync(enrollmentId, {
    fields: [CONFIG.enrollmentFields.athlete],
  });

  if (!enrollmentRecord) {
    throw new Error(`Enrollment record not found: ${enrollmentId}`);
  }

  const athleteId = getLinkedIds(enrollmentRecord, CONFIG.enrollmentFields.athlete)[0] || "";
  if (!athleteId) {
    throw new Error(`Enrollment ${enrollmentId} is missing linked Athlete.`);
  }

  const isDryRun = getBooleanish(scenarioRecord, CONFIG.testingScenarioFields.dryRun, false);

  if (scenarioType === CONFIG.scenarioTypes.dailySubmission) {
    await runDailySubmissionBranch({
      testingScenariosTable,
      submissionsTable,
      scenarioRecord,
      recordId,
      enrollmentId,
      athleteId,
      activityDate,
      isDryRun,
      debugStep,
    });
    return;
  }

  if (scenarioType === CONFIG.scenarioTypes.homework) {
    await runHomeworkBranch({
      testingScenariosTable,
      submissionsTable,
      scenarioRecord,
      recordId,
      enrollmentId,
      athleteId,
      activityDate,
      isDryRun,
      debugStep,
    });
    return;
  }

  if (isVideoScenarioType(scenarioType)) {
    await runVideoBranch({
      testingScenariosTable,
      submissionsTable,
      scenarioRecord,
      recordId,
      enrollmentId,
      athleteId,
      activityDate,
      isDryRun,
      debugStep,
    });
    return;
  }
}

/* =========================================================
   SECTION 7 — RUN
========================================================= */

try {

} catch (error) {

  setOutputSafe("statusOut", CONFIG.statuses.error);

  setOutputSafe("errorOut", message);
  console.log(
    JSON.stringify(
      {
        automation: SCRIPT.scriptName,
        version: SCRIPT.version,
        statusOut: CONFIG.statuses.error,
        errorOut: message,
      },
      null,
      2
    )
  );
  throw error;
}
